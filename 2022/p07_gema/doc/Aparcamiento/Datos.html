def initialize(accesibilidad, seguridad, id, nombre, descripcion, tipo, conjunto_libreuocupadas, conjunto_plazas)
  @accesibilidad = accesibilidad
  @seguridad = seguridad
  @id = id
  @nombre = nombre
  @descripcion = descripcion
  @tipo = tipo
  @conjunto_libreuocupadas = conjunto_libreuocupadas
  @conjunto_plazas = conjunto_plazas

  @accesibilidad = if accesibilidad >= 1 && accesibilidad <= 5
    accesibilidad
  else
    "Accesibilidad no valido [1..5]"
  end

  @seguridad = if seguridad >= 1 && seguridad <= 10
    seguridad
  else
    "Seguridad no valido [1..10]"
  end
end

 def numero_plazas_libres
